1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The node class
  8  *
  9  * @param {object} object The node object.
 10  * @param {function} callback The function to be called once the node has
 11  * been retrieved from the server.
 12  */
 13 drupal.node = function(object, callback) {
 14   drupal.entity.call(this, object, callback);
 15 };
 16 
 17 /** Derive from entity. */
 18 drupal.node.prototype = new drupal.entity();
 19 
 20 /** Reset the constructor. */
 21 drupal.node.prototype.constructor = drupal.node;
 22 
 23 /**
 24  * Sets the object.
 25  *
 26  * @param {object} object The object which contains the data.
 27  */
 28 drupal.node.prototype.set = function(object) {
 29   drupal.entity.prototype.set.call(this, object);
 30 
 31   /** Set the api. */
 32   this.api = this.api || new drupal.node.api();
 33 
 34   /** Set the ID based on the nid. */
 35   this.id = object.nid || this.id;
 36 
 37   /** The title for this node. */
 38   this.title = object.title || '';
 39 
 40   /** The type of node we are dealing with. */
 41   this.type = object.type || '';
 42 
 43   /** The status of this node. */
 44   this.status = object.status || 0;
 45 
 46   /** The user who created this node */
 47   this.uid = object.uid || 0;
 48 };
 49 
 50 /**
 51  * Returns the object to send to Services.
 52  *
 53  * @return {object} The object to send to the Services endpoint.
 54  */
 55 drupal.node.prototype.get = function() {
 56   return jQuery.extend(drupal.entity.prototype.get.call(this), {
 57     title: this.title,
 58     type: this.type,
 59     status: this.status,
 60     uid: this.uid
 61   });
 62 };
 63 
 64 /**
 65  * Override the setQuery method of the entity.
 66  *
 67  * @param {object} query The query object.
 68  * @param {string} param The param to set.
 69  * @param {string} value The value of the field to set.
 70  */
 71 drupal.node.prototype.setQuery = function(query, param, value) {
 72   // The node object sets parameters like ?parameters[param]=value...
 73   query['parameters[' + param + ']'] = value;
 74 };
