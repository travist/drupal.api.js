1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /** The current logged in user. */
  5 drupal.current_user = null;
  6 
  7 /**
  8  * @constructor
  9  * @extends drupal.entity
 10  * @class The user class
 11  *
 12  * @param {object} object The user object.
 13  * @param {function} callback The function to be called once the user has
 14  * been retrieved from the server.
 15  * @param {object} options Options used to govern functionality.
 16  */
 17 drupal.user = function(object, callback, options) {
 18   drupal.entity.call(this, object, callback, options);
 19 };
 20 
 21 /** Derive from drupal.entity. */
 22 drupal.user.prototype = new drupal.entity();
 23 
 24 /** Reset the constructor. */
 25 drupal.user.prototype.constructor = drupal.user;
 26 
 27 /** Declare the user api. */
 28 drupal.user.api = jQuery.extend(new drupal.api(), {
 29   resource: 'user'
 30 });
 31 
 32 /**
 33  * Returns an index of users.
 34  *
 35  * @param {object} query The query parameters.
 36  * @param {function} callback The callback function.
 37  * @param {object} options Options used to govern functionality.
 38  */
 39 drupal.user.index = function(query, callback, options) {
 40   drupal.entity.index(drupal.user, query, callback, options);
 41 };
 42 
 43 /**
 44  * Sets the object.
 45  *
 46  * @param {object} object The object which contains the data.
 47  */
 48 drupal.user.prototype.set = function(object) {
 49   drupal.entity.prototype.set.call(this, object);
 50 
 51   /** The name of this entity. */
 52   this.entityName = 'user';
 53 
 54   /** Set the api. */
 55   this.api = drupal.user.api;
 56 
 57   /** Set the ID based on the uid. */
 58   this.id = object.uid || this.id || 0;
 59 
 60   /** The name for this user. */
 61   this.name = object.name || this.name || '';
 62 
 63   /** The email address of our user. */
 64   this.mail = object.mail || this.mail || '';
 65 
 66   /** The password of the user. */
 67   this.pass = object.pass || this.pass || '';
 68 
 69   /** The status of the user. */
 70   this.status = object.status || this.status || 1;
 71 };
 72 
 73 /**
 74  * Sets a user session.
 75  *
 76  * @param {string} name The name of the session.
 77  * @param {string} sessid The session ID.
 78  */
 79 drupal.user.prototype.setSession = function(name, sessid) {
 80 
 81   /** Set the session id for this user. */
 82   this.sessid = sessid;
 83 
 84   // Only set the session name if this user is valid and has a session name.
 85   if (this.id && name) {
 86 
 87     /** Set the session name for this user. */
 88     this.session_name = name;
 89 
 90     /** Now store this in a cookie for further authentication. */
 91     drupal.cookie(name, sessid);
 92 
 93     // Now store this user as the 'current' user.
 94     drupal.current_user = this;
 95   }
 96 };
 97 
 98 /**
 99  * Login a user.
100  *
101  * @param {function} callback The callback function.
102  */
103 drupal.user.prototype.login = function(callback) {
104   if (this.api) {
105     this.api.execute('login', {
106       username: this.name,
107       password: this.pass
108     }, (function(user) {
109       return function(object) {
110 
111         // Update this object.
112         user.update(object.user);
113 
114         // Set the session.
115         user.setSession(object.session_name, object.sessid);
116 
117         if (callback) {
118           callback.call(user, user);
119         }
120       };
121     })(this));
122   }
123 };
124 
125 /**
126  * Register a user.
127  *
128  * @param {function} callback The callback function.
129  */
130 drupal.user.prototype.register = function(callback) {
131   if (this.api) {
132     this.api.execute('register', this.getPOST(), (function(user) {
133       return function(object) {
134         user.update(object, callback);
135       };
136     })(this));
137   }
138 };
139 
140 /**
141  * Logout the user.
142  *
143  * @param {function} callback The callback function.
144  */
145 drupal.user.prototype.logout = function(callback) {
146   if (this.api) {
147     this.api.execute('logout', null, callback);
148   }
149 };
150 
151 /**
152  * Gets a POST object.
153  *
154  * @return {object} The filtered object.
155  */
156 drupal.user.prototype.getPOST = function() {
157 
158   // Add the password to POST's only.
159   var post = drupal.entity.prototype.getPOST.call(this);
160   post.pass = this.pass;
161   return post;
162 };
163 
164 /**
165  * Returns the object to send to Services.
166  *
167  * @return {object} The object to send to the Services endpoint.
168  */
169 drupal.user.prototype.get = function() {
170   return jQuery.extend(drupal.entity.prototype.get.call(this), {
171     name: this.name,
172     mail: this.mail,
173     status: this.status
174   });
175 };
