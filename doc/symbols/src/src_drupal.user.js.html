1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /** The current logged in user. */
  5 drupal.current_user = null;
  6 
  7 /**
  8  * @constructor
  9  * @extends drupal.entity
 10  * @class The user class
 11  *
 12  * @param {object} object The user object.
 13  * @param {function} callback The function to be called once the user has
 14  * been retrieved from the server.
 15  */
 16 drupal.user = function(object, callback) {
 17   drupal.entity.call(this, object, callback);
 18 };
 19 
 20 /** Derive from drupal.entity. */
 21 drupal.user.prototype = new drupal.entity();
 22 
 23 /** Reset the constructor. */
 24 drupal.user.prototype.constructor = drupal.user;
 25 
 26 /**
 27  * Sets the object.
 28  *
 29  * @param {object} object The object which contains the data.
 30  */
 31 drupal.user.prototype.set = function(object) {
 32   drupal.entity.prototype.set.call(this, object);
 33 
 34   /** The name of this entity. */
 35   this.entityName = 'user';
 36 
 37   /** Set the api. */
 38   this.api = this.api || new drupal.user.api();
 39 
 40   /** Set the ID based on the uid. */
 41   this.id = object.uid || this.id || 0;
 42 
 43   /** The name for this user. */
 44   this.name = object.name || this.name || '';
 45 
 46   /** The email address of our user. */
 47   this.mail = object.mail || this.mail || '';
 48 
 49   /** The password of the user. */
 50   this.pass = object.pass || this.pass || '';
 51 
 52   /** The status of the user. */
 53   this.status = object.status || this.status || 1;
 54 };
 55 
 56 /**
 57  * Sets a user session.
 58  *
 59  * @param {string} name The name of the session.
 60  * @param {string} sessid The session ID.
 61  */
 62 drupal.user.prototype.setSession = function(name, sessid) {
 63 
 64   /** Set the session id for this user. */
 65   this.sessid = sessid;
 66 
 67   // Only set the session name if this user is valid and has a session name.
 68   if (this.id && name) {
 69 
 70     /** Set the session name for this user. */
 71     this.session_name = name;
 72 
 73     /** Now store this in a cookie for further authentication. */
 74     drupal.cookie(name, sessid);
 75 
 76     /** Now store this user as the 'current' user. */
 77     drupal.current_user = this;
 78   }
 79 };
 80 
 81 /**
 82  * Login a user.
 83  *
 84  * @param {function} callback The callback function.
 85  */
 86 drupal.user.prototype.login = function(callback) {
 87   if (this.api) {
 88     this.api.execute('login', {
 89       username: this.name,
 90       password: this.pass
 91     }, (function(user) {
 92       return function(object) {
 93 
 94         // Set the session.
 95         user.setSession(object.session_name, object.sessid);
 96 
 97         // Update this object.
 98         user.update(object.user, callback);
 99       };
100     })(this));
101   }
102 };
103 
104 /**
105  * Register a user.
106  *
107  * @param {function} callback The callback function.
108  */
109 drupal.user.prototype.register = function(callback) {
110   if (this.api) {
111     this.api.execute('register', this.getPOST(), (function(user) {
112       return function(object) {
113         user.update(object, callback);
114       };
115     })(this));
116   }
117 };
118 
119 /**
120  * Logout the user.
121  *
122  * @param {function} callback The callback function.
123  */
124 drupal.user.prototype.logout = function(callback) {
125   if (this.api) {
126     this.api.execute('logout', null, callback);
127   }
128 };
129 
130 /**
131  * Gets a POST object.
132  *
133  * @return {object} The filtered object.
134  */
135 drupal.user.prototype.getPOST = function() {
136 
137   // Add the password to POST's only.
138   var post = drupal.entity.prototype.getPOST.call(this);
139   post.pass = this.pass;
140   return post;
141 };
142 
143 /**
144  * Returns the object to send to Services.
145  *
146  * @return {object} The object to send to the Services endpoint.
147  */
148 drupal.user.prototype.get = function() {
149   return jQuery.extend(drupal.entity.prototype.get.call(this), {
150     name: this.name,
151     mail: this.mail,
152     status: this.status
153   });
154 };
