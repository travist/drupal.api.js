1 /** The drupal namespace. */
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base Drupal API class.
  7  */
  8 drupal.api = function() {
  9 
 10   /** The Services API endpoint */
 11   this.endpoint = drupal.endpoint || this.endpoint || '';
 12 
 13   /** The resource within this endpoint */
 14   this.resource = this.resource || '';
 15 };
 16 
 17 /**
 18  * Helper function to get the Services URL for this resource.
 19  *
 20  * @param {object} object The object involved with in this request.
 21  * @return {string} The path to the API endpoint.
 22  */
 23 drupal.api.prototype.getURL = function(object) {
 24   var path = this.endpoint;
 25   path += this.resource ? ('/' + this.resource) : '';
 26   path += (object && object.id) ? ('/' + object.id) : '';
 27   return path;
 28 };
 29 
 30 /**
 31  * API function to act as a generic request for all Service calls.
 32  *
 33  * @param {string} url The URL where the request will go.
 34  * @param {string} dataType The type of request.  json or jsonp.
 35  * @param {string} type The type of HTTP request.  GET, POST, PUT, etc.
 36  * @param {object} data The data to send to the server.
 37  * @param {function} callback The function callback.
 38  */
 39 drupal.api.prototype.call = function(url, dataType, type, data, callback) {
 40   var request = {
 41     url: url,
 42     dataType: dataType,
 43     type: type,
 44     success: function(data, textStatus) {
 45       if (textStatus == 'success') {
 46         if (callback) {
 47           callback(data);
 48         }
 49       }
 50       else {
 51         console.log('Error: ' + textStatus);
 52       }
 53     },
 54     error: function(xhr, ajaxOptions, thrownError) {
 55       console.log(xhr.responseText);
 56       if (callback) {
 57         callback(null);
 58       }
 59     }
 60   };
 61 
 62   if (data) {
 63     request['data'] = data;
 64   }
 65 
 66   // Make the request.
 67   jQuery.ajax(request);
 68 };
 69 
 70 /**
 71  * API function to get any results from the drupal API.
 72  *
 73  * @param {object} object The object of the item we are getting..
 74  * @param {object} query key-value pairs to add to the query of the URL.
 75  * @param {function} callback The callback function.
 76  */
 77 drupal.api.prototype.get = function(object, query, callback) {
 78   var url = this.getURL(object);
 79   url += '.jsonp';
 80   url += query ? ('?' + decodeURIComponent(jQuery.param(query, true))) : '';
 81   this.call(url, 'jsonp', 'GET', null, callback);
 82 };
 83 
 84 /**
 85  * API function to get a type of object within an object.
 86  *
 87  * @param {object} object The object of the item we are getting..
 88  * @param {string} type The type of object you wish to get within this object.
 89  * @param {object} query key-value pairs to add to the query of the URL.
 90  * @param {function} callback The callback function.
 91  */
 92 drupal.api.prototype.getItems = function(object, type, query, callback) {
 93   var url = this.getURL(object) + '/' + type;
 94   url += '.jsonp';
 95   url += query ? ('?' + decodeURIComponent(jQuery.param(query, true))) : '';
 96   this.call(url, 'jsonp', 'GET', null, callback);
 97 };
 98 
 99 /**
100  * API function to perform an action.
101  *
102  * @param {string} action The action to perform.
103  * @param {object} object The entity object to set.
104  * @param {function} callback The callback function.
105  */
106 drupal.api.prototype.execute = function(action, object, callback) {
107   var url = this.getURL(object) + '/' + action;
108   this.call(url, 'json', 'POST', object, callback);
109 };
110 
111 /**
112  * API function to save the value of an object using Services.
113  *
114  * @param {object} object The entity object to set.  If the object does not
115  * have an ID, then this will create a new entity, otherwise, it will simply
116  * update the existing resource.
117  *
118  * @param {function} callback The callback function.
119  *
120  **/
121 drupal.api.prototype.save = function(object, callback) {
122   var type = object.id ? 'PUT' : 'POST';
123   this.call(this.getURL(object), 'json', type, object, callback);
124 };
125 
126 /**
127  * API function to remove an object on the server.
128  *
129  * @param {object} object The entity object to delete.
130  * @param {function} callback The callback function.
131  */
132 drupal.api.prototype.remove = function(object, callback) {
133   this.call(this.getURL(object), 'json', 'DELETE', null, callback);
134 };
