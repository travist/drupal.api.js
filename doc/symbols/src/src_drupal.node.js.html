1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends drupal.entity
  7  * @class The node class
  8  *
  9  * @param {object} object The node object.
 10  * @param {function} callback The function to be called once the node has
 11  * been retrieved from the server.
 12  * @param {object} options Options used to govern functionality.
 13  */
 14 drupal.node = function(object, callback, options) {
 15   drupal.entity.call(this, object, callback, options);
 16 };
 17 
 18 /** Derive from entity. */
 19 drupal.node.prototype = new drupal.entity();
 20 
 21 /** Reset the constructor. */
 22 drupal.node.prototype.constructor = drupal.node;
 23 
 24 /** Declare the node api. */
 25 drupal.node.api = jQuery.extend(new drupal.api(), {
 26   resource: 'node'
 27 });
 28 
 29 /**
 30  * Returns an index of nodes.
 31  *
 32  * @param {object} query The query parameters.
 33  * @param {function} callback The callback function.
 34  * @param {object} options Options used to govern functionality.
 35  */
 36 drupal.node.index = function(query, callback, options) {
 37   drupal.entity.index(drupal.node, query, callback, options);
 38 };
 39 
 40 /**
 41  * Sets the object.
 42  *
 43  * @param {object} object The object which contains the data.
 44  */
 45 drupal.node.prototype.set = function(object) {
 46   drupal.entity.prototype.set.call(this, object);
 47 
 48   /** The name of this entity. */
 49   this.entityName = 'node';
 50 
 51   /** Set the api to the drupal.node.api. */
 52   this.api = drupal.node.api;
 53 
 54   /** Set the ID based on the nid. */
 55   this.id = object.nid || this.id || 0;
 56 
 57   // Set the values for this entity.
 58   this.setValues({
 59     title: '',
 60     type: '',
 61     status: 0,
 62     uid: 0
 63   }, object);
 64 };
 65 
 66 /**
 67  * Returns the object to send to Services.
 68  *
 69  * @return {object} The object to send to the Services endpoint.
 70  */
 71 drupal.node.prototype.get = function() {
 72   return jQuery.extend(drupal.entity.prototype.get.call(this), {
 73     title: this.title,
 74     type: this.type,
 75     status: this.status,
 76     uid: this.uid
 77   });
 78 };
 79 
 80 /**
 81  * Override the getQuery method of the entity.
 82  *
 83  * @param {object} query The query variables.
 84  * @return {object} The query variables.
 85  */
 86 drupal.node.prototype.getQuery = function(query) {
 87   query = drupal.entity.prototype.getQuery.call(this, query);
 88   if (query.type) {
 89     query['parameters[type]'] = query.type;
 90     delete query.type;
 91   }
 92   return query;
 93 };
