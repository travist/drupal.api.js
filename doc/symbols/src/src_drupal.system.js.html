1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The system class
  8    *
  9    * @param {function} callback The function to be called once the system has
 10    * connected.
 11    */
 12   drupal.system = function(callback) {
 13 
 14     /** The current user. */
 15     this.user = null;
 16 
 17     /** The session ID */
 18     this.sessid = '';
 19 
 20     // Declare the api.
 21     this.api = new drupal.system.api();
 22 
 23     // If the callback is set, then connect.
 24     if (callback) {
 25 
 26       // Connect to the server.
 27       this.connect(callback);
 28     }
 29   };
 30 
 31   /**
 32    * Connect to the server.
 33    *
 34    * @param {function} callback The callback function.
 35    */
 36   drupal.system.prototype.connect = function(callback) {
 37 
 38     // Connect to the server.
 39     var _this = this;
 40     this.api.execute('connect', null, function(object) {
 41 
 42       // Set the user object, session id, and return this server.
 43       _this.user = new drupal.user(object.user);
 44       _this.sessid = object.sessid;
 45       callback(_this);
 46     });
 47   };
 48 
 49   /**
 50    * Get a variable from the server.
 51    *
 52    * @param {string} name The variable you wish to retrieve.
 53    * @param {string} def The default value of the variable.
 54    * @param {function} callback The callback function.
 55    */
 56   drupal.system.prototype.get_variable = function(name, def, callback) {
 57     this.api.execute('get_variable', {name: name, 'default': def}, callback);
 58   };
 59 
 60   /**
 61    * Set a variable on the server.
 62    *
 63    * @param {string} name The variable you wish to set.
 64    * @param {string} value The value of the variable.
 65    * @param {function} callback The callback function.
 66    */
 67   drupal.system.prototype.set_variable = function(name, value, callback) {
 68     this.api.execute('set_variable', {name: name, value: value}, callback);
 69   };
 70 
 71   /**
 72    * Delete a variable on the server.
 73    *
 74    * @param {string} name The variable you wish to set.
 75    * @param {function} callback The callback function.
 76    */
 77   drupal.system.prototype.del_variable = function(name, callback) {
 78     this.api.execute('del_variable', {name: name}, callback);
 79   };
 80 }(jQuery));
