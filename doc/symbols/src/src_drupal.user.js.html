1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The user class
  8    *
  9    * @extends drupal.entity
 10    *
 11    * @param {object} object The user object.
 12    * @param {function} callback The function to be called once the user has
 13    * been retrieved from the server.
 14    */
 15   drupal.user = function(object, callback) {
 16 
 17     /** The name for this user. */
 18     this.name = '';
 19 
 20     /** The email address of our user. */
 21     this.mail = '';
 22 
 23     /** The password of the user. */
 24     this.pass = '';
 25 
 26     // Declare the api.
 27     this.api = new drupal.user.api();
 28 
 29     // Call the base class.
 30     drupal.entity.call(this, object, callback);
 31   };
 32 
 33   // Derive from entity.
 34   drupal.user.prototype = new drupal.entity();
 35   drupal.user.prototype.constructor = drupal.user;
 36 
 37   /**
 38    * Login a user.
 39    *
 40    * @param {function} callback The callback function.
 41    */
 42   drupal.user.prototype.login = function(callback) {
 43 
 44     // Setup the POST data for the login of this user.
 45     var object = {
 46       username: this.name,
 47       password: this.pass
 48     };
 49 
 50     // Execute the login.
 51     var _this = this;
 52     this.api.execute('login', object, function(user) {
 53 
 54       // Update this object.
 55       _this.update(user);
 56       callback(_this);
 57     });
 58   };
 59 
 60   /**
 61    * Register a user.
 62    *
 63    * @param {function} callback The callback function.
 64    */
 65   drupal.user.prototype.register = function(callback) {
 66 
 67     // Setup the POST data to register this user.
 68     var object = {
 69       name: this.name,
 70       mail: this.mail,
 71       pass: this.pass
 72     };
 73 
 74     // Execute the register.
 75     var _this = this;
 76     this.api.execute('register', object, function(user) {
 77 
 78       // Update this object.
 79       _this.update(user);
 80       callback(_this);
 81     });
 82   };
 83 
 84   /**
 85    * Logout the user.
 86    */
 87   drupal.user.prototype.logout = function(callback) {
 88 
 89     // Execute the logout.
 90     this.api.execute('logout', null, callback);
 91   };
 92 
 93   /**
 94    * Override the update routine.
 95    *
 96    * @param {object} object The object to update.
 97    */
 98   drupal.user.prototype.update = function(object) {
 99 
100     drupal.entity.prototype.update.call(this, object);
101 
102     // Make sure to also set the ID the same as uid.
103     this.id = object.uid || this.id;
104   };
105 
106   /**
107    * Returns the object to send to Services.
108    */
109   drupal.user.prototype.getObject = function() {
110     return $.extend(drupal.entity.prototype.getObject.call(this), {
111       name: this.name,
112       mail: this.mail,
113       pass: this.pass
114     });
115   };
116 }(jQuery));
