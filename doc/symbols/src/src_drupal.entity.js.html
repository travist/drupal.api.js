1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The base entity class to store the data that is common to all
  8    * drupal entities whether it be groups, events, users, etc.
  9    *
 10    * @param {object} object The entity object.
 11    */
 12   drupal.entity = function(object, callback) {
 13 
 14     /** The unique identifier for this entity. */
 15     this.id = '';
 16 
 17     /** The API for this entity */
 18     this.api = this.api || null;
 19 
 20     // If object is a string, assume it is a UUID and get it.
 21     this.update(object);
 22 
 23     // If they provide a callback, call it now.
 24     if (callback) {
 25 
 26       // Get the object from the server.
 27       this.get(callback);
 28     }
 29   };
 30 
 31   /**
 32    *
 33    */
 34   drupal.entity.prototype.find = function(callback) {
 35 
 36 
 37   };
 38 
 39   /**
 40    * Get's an object from the drupal API.
 41    *
 42    * @param {function} callback The callback function when the object is
 43    * retrieved.
 44    */
 45   drupal.entity.prototype.get = function(callback) {
 46 
 47     // If the API exists, then we need to get the object.
 48     if (this.api) {
 49 
 50       // Call the API.
 51       var _this = this;
 52       this.api.get(this.getObject(), this.getQuery(), function(object) {
 53 
 54         // If this is an array, then just return the results.
 55         if (object[0]) {
 56           callback(object);
 57         }
 58         else {
 59           // Update the object, then call the callback.
 60           _this.update(object);
 61           callback(_this);
 62         }
 63       });
 64     }
 65   };
 66 
 67   /**
 68    * Saves this entity.
 69    *
 70    * @param {function} callback The function called once entity is saved.
 71    */
 72   drupal.entity.prototype.save = function(callback) {
 73 
 74     // If the API exists, then we can save this object.
 75     if (this.api) {
 76 
 77       // Call the API.
 78       var _this = this;
 79       this.api.save(this.getObject(), function(object) {
 80 
 81         // Update the object, then call the callback.
 82         _this.update(object);
 83         callback(_this);
 84       });
 85     }
 86   };
 87 
 88   /**
 89    * Removes an entity
 90    *
 91    * @param {function} callback The function called once entity is removed.
 92    */
 93   drupal.entity.prototype.remove = function(callback) {
 94 
 95     // Only remove if they have an ID.
 96     if (this.id) {
 97 
 98       // Call the API.
 99       this.api.remove(this.getObject(), callback);
100     }
101   };
102 
103   /**
104    * Returns the search query.
105    */
106   drupal.entity.prototype.getQuery = function() {
107 
108     var query = {};
109 
110     // We only need to provide a search query if there is no ID.
111     if (!this.id) {
112 
113       // Iterate through all of our fields.
114       for (var field in this) {
115         if (this.hasOwnProperty(field) &&
116             this[field] &&
117             (typeof this[field] != 'object')) {
118 
119           // Add this as a search parameter.
120           query['parameters[' + field + ']'] = this[field];
121         }
122       }
123     }
124 
125     // Return the params.
126     return query;
127   };
128 
129   /**
130    * Update the entity data.
131    *
132    * @param {object} entity The entity information.
133    */
134   drupal.entity.prototype.update = function(object) {
135 
136     // Update the object.
137     if (object) {
138 
139       // Update the params.
140       for (var param in object) {
141         if (object.hasOwnProperty(param) && this.hasOwnProperty(param)) {
142 
143           // Check to see if this object has an update function.
144           if (this[param].update) {
145             this[param].update(object[param]);
146           }
147           else {
148             this[param] = object[param];
149           }
150         }
151       }
152     }
153   };
154 
155   /**
156    * Returns the object to send during PUT's and POST's during a save or add.
157    *
158    * @return {object} An object of the data when saving to the server.
159    */
160   drupal.entity.prototype.getObject = function() {
161     return {
162       id: this.id
163     };
164   };
165 }(jQuery));
