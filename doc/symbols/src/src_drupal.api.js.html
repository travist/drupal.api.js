1 /** The drupal namespace. */
  2 var drupal = drupal || {};
  3 
  4 /** Determine if we have storage. */
  5 drupal.hasStorage = (typeof(Storage) !== 'undefined');
  6 drupal.hasStorage &= (typeof(JSON) !== 'undefined');
  7 
  8 /**
  9  * Retrieve an item out of local storage.
 10  *
 11  * @param {string} key The key for the object to retrieve.
 12  * @return {object} The object that was retrieved.
 13  */
 14 drupal.retrieve = function(key) {
 15   var object = null;
 16 
 17   // Check to see if we have storage.
 18   if (key && drupal.hasStorage) {
 19 
 20     // Get it out of localStorage.
 21     if (object = JSON.parse(localStorage.getItem(key))) {
 22 
 23       // Make sure this object hasn't expired.
 24       if ((new Date()).getTime() > object.expires) {
 25 
 26         // Clear it if it has.
 27         localStorage.removeItem(key);
 28         object = {};
 29       }
 30     }
 31   }
 32 
 33   return object;
 34 };
 35 
 36 /**
 37  * Store an object with an expiration.
 38  *
 39  * @param {string} key The key for the object to store.
 40  * @param {object} object They object to store.
 41  * @param {number} expires The expiration (in seconds) for this object.
 42  */
 43 drupal.store = function(key, object, expires) {
 44 
 45   // Default the expiration if it wasn't provided.
 46   expires = expires || 3600;
 47 
 48   // Make sure we can store.
 49   if (key && drupal.hasStorage) {
 50 
 51     // Set an expiration date for this object.
 52     object.expires = (expires * 1000) + (new Date()).getTime();
 53 
 54     // Store this object in localStorage.
 55     localStorage.setItem(key, JSON.stringify(object));
 56   }
 57 };
 58 
 59 /**
 60  * Clears an object out of storage.
 61  *
 62  * @param {string} key The key for this object to clear.
 63  */
 64 drupal.clear = function(key) {
 65   if (key && drupal.hasStorage) {
 66     localStorage.removeItem(key);
 67   }
 68 };

 70 /**
 71  * The Drupal API class.  This is a static class helper
 72  * to assist in communication between javascript and
 73  * a Drupal CMS backend.
 74  *
 75  * @return {object} The API object.
 76  */
 77 drupal.api = function() {
 78   return {
 79 
 80     /** The resource within this endpoint */
 81     resource: '',
 82     cacheId: '',
 83 
 84     /** See if we are dealing with jQuery Mobile applications. */
 85     isMobile: jQuery.hasOwnProperty('mobile'),
 86 
 87     /**
 88      * The Services API endpoint
 89      *
 90      * @this {object} The drupal.api object.
 91      * @return {string} The services endpoint.
 92      **/
 93     endpoint: function() {
 94       return drupal.endpoint || '';
 95     },
 96 
 97     /**
 98      * Helper function to get the Services URL for this resource.
 99      *
100      * @this {object} The drupal.api object.
101      * @param {object} object The object involved with in this request.
102      * @return {string} The path to the API endpoint.
103      */
104     getURL: function(entity) {
105       // If the entity has a valid URI, then use that...
106       if (entity && entity.uri) {
107         return entity.uri;
108       }
109       else {
110 
111         // Otherwise, build our best guess for the URI of this entity.
112         var path = this.endpoint();
113         path += this.resource ? ('/' + this.resource) : '';
114         path += (entity && entity.id) ? ('/' + entity.id) : '';
115         return path;
116       }
117     },
118 
119     /**
120      * Called when we are loading or not.
121      *
122      * @param {boolean} loading If this api is loading something.
123      * @this Points to the drupal.api object.
124      */
125     loading: function(loading) {
126       if (this.isMobile) {
127         if (loading) {
128           jQuery('body').addClass('ui-loading');
129         }
130         else {
131           jQuery('body').removeClass('ui-loading');
132         }
133       }
134     },
135 
136     /**
137      * API function to act as a generic request for all Service calls.
138      *
139      * @this {object} The drupal.api object.
140      * @param {string} url The URL where the request will go.
141      * @param {string} dataType The type of request.  json or jsonp.
142      * @param {string} type The type of HTTP request.  GET, POST, PUT, etc.
143      * @param {object} data The data to send to the server.
144      * @param {function} callback The function callback.
145      */
146     call: function(url, dataType, type, data, callback) {
147       var request = {
148         url: url,
149         dataType: dataType,
150         type: type,
151         success: (function(api) {
152           return function(data, textStatus) {
153             api.loading(false);
154             if (textStatus == 'success') {
155               if (callback) {
156                 callback(data);
157               }
158             }
159             else {
160               console.log('Error: ' + textStatus);
161             }
162           };
163         })(this),
164         error: (function(api) {
165           return function(xhr, ajaxOptions, thrownError) {
166             api.loading(false);
167             console.log(xhr.responseText);
168             if (callback) {
169               callback(null);
170             }
171           };
172         })(this)
173       };
174 
175       if (data) {
176         request['data'] = data;
177       }
178 
179       // Show a loading cursor.
180       this.loading(true);
181 
182       // Make the request.
183       jQuery.ajax(request);
184     },
185 
186     /**
187      * API function to get any results from the drupal API.
188      *
189      * Return the object.
190      *
191      *  drupal.api.get(entity, function(object) {
192      *    console.log(object);
193      *  });
194      *
195      * Return a list of events within an entity.
196      *
197      *  drupal.api.get(entity, 'events', function(events) {
198      *    console.log(events);
199      *  });
200      *
201      * Return a list of nodes with type='page'.
202      *
203      *  drupal.api.get({}, {type:'page'}, function(object) {
204      *
205      *  });
206      *
207      * Return a list of events provided a query within a node.
208      *
209      *  drupal.api.get(entity, 'events', {month: 6}, functoin(events) {
210      *    console.log(events);
211      *  });
212      *
213      *
214      * @this {object} The drupal.api object.
215      * @param {object} object The object of the item we are getting..
216      * @param {string} endpoint An additional endpoint to add onto the resource.
217      * @param {object} query key-value pairs to add to the query of the URL.
218      * @param {function} callback The callback function.
219      * @param {boolean} cache cache/get the results in/from localStorage.
220      */
221     get: function(object, endpoint, query, callback, cache) {
222       // Normalize the arguments based on the different schemes of calling this.
223       var type = (typeof endpoint);
224       if (type === 'object') {
225         callback = query;
226         query = endpoint;
227         endpoint = '';
228       }
229       else if (type === 'function') {
230         callback = endpoint;
231         endpoint = '';
232       }
233 
234       // Get the url for this object.
235       var url = this.getURL(object);
236       url += (endpoint) ? ('/' + endpoint) : '';
237       url += '.jsonp';
238       url += query ? ('?' + decodeURIComponent(jQuery.param(query, true))) : '';
239 
240       // See if we should cache the result.
241       if (cache) {
242         this.cacheId = this.cacheId || url.replace(/[^A-z0-9]/g, '');
243         var storage = drupal.retrieve(cacheId);
244         if (storage) {
245           callback(storage);
246           return;
247         }
248       }
249 
250       // No cache exists, so make the server call.
251       this.call(url, 'jsonp', 'GET', null, (function(api) {
252         return function(data) {
253 
254           // Store this in cache...
255           if (cache) {
256             drupal.store(api.cacheId, data);
257           }
258 
259           // Store the result.
260           callback(data);
261         }
262       })(this));
263     },
264 
265     /**
266      * API function to perform an action.
267      *
268      * @this {object} The drupal.api object.
269      * @param {string} action The action to perform.
270      * @param {object} object The entity object to set.
271      * @param {function} callback The callback function.
272      */
273     execute: function(action, object, callback) {
274       var url = this.getURL(object) + '/' + action;
275       this.call(url, 'json', 'POST', object, callback);
276     },
277 
278     /**
279      * API function to save the value of an object using Services.
280      *
281      * @this {object} The drupal.api object.
282      * @param {object} object The entity object to set.  If the object does not
283      * have an ID, then this will create a new entity, otherwise, it will simply
284      * update the existing resource.
285      *
286      * @param {function} callback The callback function.
287      *
288      */
289     save: function(object, callback) {
290       var type = object.id ? 'PUT' : 'POST';
291       this.call(this.getURL(object), 'json', type, object, callback);
292     },
293 
294     /**
295      * API function to remove an object on the server.
296      *
297      * @this {object} The drupal.api object.
298      * @param {object} object The entity object to delete.
299      * @param {function} callback The callback function.
300      */
301     remove: function(object, callback) {
302 
303       // Remove the storage if the cacheID exists.
304       if (this.cacheId) {
305         drupal.clear(this.cacheId);
306       }
307 
308       // Call to delete the resource.
309       this.call(this.getURL(object), 'json', 'DELETE', null, callback);
310     }
311   };
312 };
