1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base entity class to store the data that is common to all
  7  * drupal entities whether it be groups, events, users, etc.
  8  *
  9  * @param {object} object The entity object.
 10  * @param {function} callback The callback function to get the object.
 11  */
 12 drupal.entity = function(object, callback) {
 13 
 14   /** The unique identifier for this entity. */
 15   this.id = '';
 16 
 17   /** The API for this entity */
 18   this.api = this.api || null;
 19 
 20   // If object is a string, assume it is a UUID and get it.
 21   this.update(object);
 22 
 23   // If they provide a callback, call it now.
 24   if (callback) {
 25 
 26     // Get the object from the server.
 27     this.get(callback);
 28   }
 29 };
 30 
 31 /**
 32  * Get's an object from the drupal API.
 33  *
 34  * @param {function} callback The callback function when the object is
 35  * retrieved.
 36  */
 37 drupal.entity.prototype.get = function(callback) {
 38 
 39   // If the API exists, then we need to get the object.
 40   if (this.api) {
 41 
 42     // Call the API.
 43     var _this = this;
 44     this.api.get(this.getObject(), this.getQuery(), function(object) {
 45 
 46       // If this is an array, then just return the results.
 47       if (object[0]) {
 48         callback(object);
 49       }
 50       else {
 51         // Update the object, then call the callback.
 52         _this.update(object);
 53         callback(_this);
 54       }
 55     });
 56   }
 57 };
 58 
 59 /**
 60  * Saves this entity.
 61  *
 62  * @param {function} callback The function called once entity is saved.
 63  */
 64 drupal.entity.prototype.save = function(callback) {
 65 
 66   // If the API exists, then we can save this object.
 67   if (this.api) {
 68 
 69     // Call the API.
 70     var _this = this;
 71     this.api.save(this.getObject(), function(object) {
 72 
 73       // Update the object, then call the callback.
 74       _this.update(object);
 75       callback(_this);
 76     });
 77   }
 78 };
 79 
 80 /**
 81  * Removes an entity
 82  *
 83  * @param {function} callback The function called once entity is removed.
 84  */
 85 drupal.entity.prototype.remove = function(callback) {
 86 
 87   // Only remove if they have an ID.
 88   if (this.id) {
 89 
 90     // Call the API.
 91     this.api.remove(this.getObject(), callback);
 92   }
 93 };
 94 
 95 /**
 96  * Returns the search query.
 97  *
 98  * @return {object} The query to pass to the server.
 99  */
100 drupal.entity.prototype.getQuery = function() {
101 
102   var query = {};
103 
104   // We only need to provide a search query if there is no ID.
105   if (!this.id) {
106 
107     // Iterate through all of our fields.
108     for (var field in this) {
109       if (this.hasOwnProperty(field) &&
110           this[field] &&
111           (typeof this[field] != 'object')) {
112 
113         // Add this as a search parameter.
114         query['parameters[' + field + ']'] = this[field];
115       }
116     }
117   }
118 
119   // Return the params.
120   return query;
121 };
122 
123 /**
124  * Update the entity data.
125  *
126  * @param {object} object The entity information.
127  */
128 drupal.entity.prototype.update = function(object) {
129 
130   // Update the object.
131   if (object) {
132 
133     // Update the params.
134     for (var param in object) {
135       if (object.hasOwnProperty(param) && this.hasOwnProperty(param)) {
136 
137         // Check to see if this object has an update function.
138         if (this[param].update) {
139           this[param].update(object[param]);
140         }
141         else {
142           this[param] = object[param];
143         }
144       }
145     }
146   }
147 };
148 
149 /**
150  * Returns the object to send during PUT's and POST's during a save or add.
151  *
152  * @return {object} The object to send to the Services endpoint.
153  */
154 drupal.entity.prototype.getObject = function() {
155   return {
156     id: this.id
157   };
158 };
